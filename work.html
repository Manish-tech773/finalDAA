Kruskal's Algorithm for Pure Water Distribution

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// Structure to represent an edge
struct Edge {
    int source, destination, weight;
};

// Function to compare edges by weight (used for sorting)
bool compareEdges(Edge a, Edge b) {
    return a.weight < b.weight;
}

// Function to find the root (or parent) of a set for a given vertex
int findParent(vector& parent, int vertex) {
    if (parent[vertex] != vertex) {
        parent[vertex] = findParent(parent, parent[vertex]); // Path compression
    }
    return parent[vertex];
}

// Function to perform union of two sets
void unionSets(vector& parent, vector& rank, int root1, int root2) {
    if (rank[root1] < rank[root2]) {
        parent[root1] = root2;
    } else if (rank[root1] > rank[root2]) {
        parent[root2] = root1;
    } else {
        parent[root2] = root1;
        rank[root1]++;
    }
}

// Kruskal's Algorithm to find the Minimum Spanning Tree (MST)
void kruskal(int vertices, vector& edges) {
    // Sort edges by weight
    sort(edges.begin(), edges.end(), compareEdges);

    // Initialize parent and rank for each vertex
    vector parent(vertices);
    vector rank(vertices, 0);
    for (int i = 0; i < vertices; i++) {
        parent[i] = i;
    }

    vector mst; // List to store edges included in the MST
    int totalCost = 0;

    // Iterate through sorted edges
    for (auto& edge : edges) {
        int root1 = findParent(parent, edge.source);
        int root2 = findParent(parent, edge.destination);

        // If adding this edge doesn't form a cycle
        if (root1 != root2) {
            mst.push_back(edge); // Add the edge to the MST
            unionSets(parent, rank, root1, root2); // Union the sets
            totalCost += edge.weight;
        }
    }

    // Print the results
    printf("Edges in the Minimum Spanning Tree:\n");
    for (auto& edge : mst) {
        printf("Zone %d -- Zone %d == Cost %d\n", edge.source, edge.destination, edge.weight);
    }
    printf("Total cost of the water distribution network: %d\n", totalCost);
}

// Main function
int main() {
    int vertices, edgesCount;

    // Input the number of zones and pipelines
    printf("Enter the number of zones (vertices): ");
    cin >> vertices;
    printf("Enter the number of pipelines (edges): ");
    cin >> edgesCount;

    vector edges;

    // Input the edges with their weights
    printf("Enter the source, destination, and cost of each pipeline:\n");
    for (int i = 0; i < edgesCount; i++) {
        int source, destination, weight;
        printf("Pipeline %d: ", i + 1);
        cin >> source >> destination >> weight;
        edges.push_back({source, destination, weight});
    }

    // Run Kruskal's algorithm
    kruskal(vertices, edges);

    return 0;
}
Prim's Algorithm for Pure Water Distribution

#include <iostream>
#include <vector>
#include <climits>

using namespace std;

// Function to find the vertex with the minimum key value not included in the MST
int findMinVertex(vector& key, vector& included, int vertices) {
    int minKey = INT_MAX, minVertex = -1;
    for (int i = 0; i < vertices; i++) {
        if (!included[i] && key[i] < minKey) {
            minKey = key[i];
            minVertex = i;
        }
    }
    return minVertex;
}

// Prim's Algorithm to find the Minimum Spanning Tree (MST)
void prims(int vertices, vector>& graph) {
    vector key(vertices, INT_MAX);  // Store the minimum cost to connect each vertex
    vector parent(vertices, -1);   // Store the parent of each vertex in the MST
    vector included(vertices, false);  // Track vertices included in the MST

    key[0] = 0;  // Start from the first vertex
    parent[0] = -1;  // The first node is the root of the MST

    for (int count = 0; count < vertices - 1; count++) {
        // Find the vertex with the minimum key value not yet included in the MST
        int u = findMinVertex(key, included, vertices);
        included[u] = true;  // Include the chosen vertex in the MST

        // Update key values of adjacent vertices
        for (int v = 0; v < vertices; v++) {
            // Update key and parent if:
            // 1. There's an edge from u to v
            // 2. v is not included in the MST
            // 3. Weight of edge (u, v) is smaller than the current key value of v
            if (graph[u][v] && !included[v] && graph[u][v] < key[v]) {
                key[v] = graph[u][v];
                parent[v] = u;
            }
        }
    }

    // Print the MST and its total cost
    printf("Edges in the Minimum Spanning Tree:\n");
    int totalCost = 0;
    for (int i = 1; i < vertices; i++) {
        printf("Zone %d -- Zone %d == Cost %d\n", parent[i], i, graph[i][parent[i]]);
        totalCost += graph[i][parent[i]];
    }
    printf("Total cost of the water distribution network: %d\n", totalCost);
}

// Main function
int main() {
    int vertices, edges;

    // Input the number of zones and pipelines
    printf("Enter the number of zones (vertices): ");
    cin >> vertices;

    // Initialize graph as an adjacency matrix
    vector> graph(vertices, vector(vertices, 0));

    printf("Enter the number of pipelines (edges): ");
    cin >> edges;

    printf("Enter the source, destination, and cost of each pipeline:\n");
    for (int i = 0; i < edges; i++) {
        int source, destination, weight;
        printf("Pipeline %d: ", i + 1);
        cin >> source >> destination >> weight;
        graph[source][destination] = weight;
        graph[destination][source] = weight;  // Since the graph is undirected
    }

    // Run Prim's algorithm
    prims(vertices, graph);

    return 0;
}
