How prim's Algorithm Is Used???

    

      Prim's Algorithm is a greedy algorithm used to find the Minimum Spanning Tree (MST) of a connected, undirected graph. The MST is a subset of edges that connects all vertices with the minimum possible total edge weight and no cycles.

// BFS Algorithm (Example Implementation)
function bfs(graph, startNode) {
  let queue = [];
  let visited = new Set();
  queue.push(startNode);
  visited.add(startNode);

  while (queue.length > 0) {
    let node = queue.shift();
    console.log(node);

    for (let neighbor of graph[node]) {
      if (!visited.has(neighbor)) {
        queue.push(neighbor);
        visited.add(neighbor);
      }
    }
  }
}
